roles
	administrador
		sinonimos: moderador, moderator
		vistas
			aprobar/rechazar la publicacion de eventos
				.fue_aprobado
			eliminar publicaciones denunciadas
				.tiene_denuncia
				.{username,fecha}[]
			notificaciones
				por cada evento por aprobar
					.tiene_aprobacion
				por cada publicacion denunciada
					.tiene_denuncia
	publicador
		sinonimos: publisher
		vistas
			add evento
				un evento es <<publicado>> si es aprobado por el administrador
					.fue_aprobado
				publicar sin aprobacion
					se habilita
						si alcanza 2 eventos consecutivos aprobados
							.eventos_publicados
							.eventos_aprobados
					se deshabilita
						si logra un evento eliminado
							.eventos_eliminados
				denegado para publicar
					si nunca ha tenido habilitada la publicacion sin aprobacion
					y si logra un evento eliminado
