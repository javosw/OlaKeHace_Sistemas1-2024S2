roles
	administrador
		sinonimos: moderador, moderator
		actividades
			<<publica>> eventos del publicador
				aprueba/rechaza dichos eventos
			elimina/oculta publicaciones reportadas
			notificaciones
				por cada evento por aprobar
				por cada publicacion denunciada
	publicador
		sinonimos: publisher
		entidad (persona,empresa) con un usuario
		actividades
			<<crea>> eventos
			@administrador: un evento es publicado si es aprobado por el administrador
			publicar sin aprobacion
				se habilita
					si alcanza 2 eventos consecutivos aprobados
				se deshabilita
					si logra un evento eliminado
			denegado para publicar
				si nunca ha tenido habilitada la publicacion sin aprobacion
				y si logra un evento eliminado
	registrado
		sinonimos: denunciador
		actividades
			notificaciones
				por cada evento por asistir
					.conteo_regresivo()
					@eventos_board cada evento debe tener un boton "deseo asistir"
			denunciar evento
				.motivos_de_denuncia[]
	visitante
		sinonimos: no registrado, anonimo
		visualiza eventos

evento
	.nombre
	.descripcion
	.publico
		adulto,infantil,arte,...
	.hashtag
		sirve para buscar eventos relacionados
	.lugar
	.fecha
	.hora
	.cupo
	.url
	.administrador

publicacion
	.evento
	.reportes
	actividades
		si reportes >= 3: <<ocultar>> publicacion

aclaraciones
	registrado vs no registrado
		el registrado puede denunciar publicaciones
		el no registrado no puede denunciar publicaciones
