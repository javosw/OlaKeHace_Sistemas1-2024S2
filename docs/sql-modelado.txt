okh:DATABASE {
	eventos:TABLE
		[PK,AI] id_evento:integer
		nombre:string
		lugar:string
		fecha:date
		hora:date
		plazas:integer
		plazas_ocupadas:integer
		descripcion:string
		link:string
		
	evento_etiquetas:TABLE
		[PK,UK] eventos.id_evento
		[PK] etiqueta:string

	evento_asistencias:TABLE
		eventos.id_evento
		clientes.username

	evento_denuncias:TABLE
		[PK] eventos.id_evento
		conteo_denuncias:number
		eliminar:bool

	revisiones_por_rutina:TABLE
		[PK] eventos.id_evento
		fue_revisado:bool

	revisiones_por_denuncia:TABLE
		[PK] eventos.id_evento
		[PK] clientes.username
		motivo:string
		fue_revisado:bool
		Administradores.username

	usuarios:TABLE
		[PK] username:string
		password

	clientes:TABLE
		[PK] Usuarios.username
		eventos_aprobados
		eventos_eliminados
		eventos_publicados
		tiene_sancion

	administradores:TABLE
		[PK] Usuarios.username
}
