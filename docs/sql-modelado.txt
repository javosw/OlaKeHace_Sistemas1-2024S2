okh:DATABASE {

	clientes:TABLE
		[PK] username:string
		password:string
		eventos_aprobados:number
		eventos_eliminados:number
		eventos_publicados:number
		tiene_sancion:bool

	eventos:TABLE
		[PK,AI] id_evento:integer
		clientes.username
		nombre:string
		lugar:string
		fecha:date
		hora:date
		plazas:integer
		plazas_ocupadas:integer
		descripcion:string
		link:string

	administradores:TABLE
		[PK] username:string
		password:string

	evento_etiquetas:TABLE
		[PK,UK] eventos.id_evento
		[PK] etiqueta:string

	evento_asistencias:TABLE
		[PK] eventos.id_evento
		[PK] clientes.username

	revisiones_por_rutina:TABLE
		[PK] eventos.id_evento
		fue_revisado:bool

	revisiones_por_denuncia:TABLE
		[PK] eventos.id_evento
		[PK] clientes.username
		motivo:string
		fue_revisado:bool

	evento_denuncias:TABLE
		[PK] eventos.id_evento
		conteo_denuncias:number
}
